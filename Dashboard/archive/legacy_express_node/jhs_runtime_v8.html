/* ── JHS RUNTIME — DETERMINISTIC STATE MACHINE ── */

const JHS_RUNTIME = (() => {
  const STATES = {
    BLOCKED: 'BLOCKED',
    PATCHED: 'PATCHED',
    IN_REVIEW: 'IN_REVIEW',
    VERSION_CONFLICT: 'VERSION_CONFLICT'
  };

  const TRANSITIONS = {
    [STATES.BLOCKED]: {
      [STATES.PATCHED]: { actor: 'HUMAN', requires: ['VALID_PATCH'] },
      [STATES.REJECTED]: { actor: 'HUMAN' }
    },
    [STATES.PATCHED]: {
      [STATES.IN_REVIEW]: { actor: 'SYSTEM', requires: ['PYTHON_ACK'] },
      [STATES.VERSION_CONFLICT]: { actor: 'SYSTEM' }
    }
  };

  const validatePatch = (patch, surface) => {
    const violations = Object.keys(patch)
      .filter(f => !surface.includes(f))
      .map(f => ({ field: f, code: 'UNAUTHORIZED_FIELD' }));
    return { valid: violations.length === 0, violations };
  };

  return { STATES, TRANSITIONS, validatePatch };
})();
